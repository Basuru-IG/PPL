
    class Math {
        const pi: float = 3.141592653589793;
        const e: float = 2.71828;
        
        func add(a: int, b: int): int {
            return a + b;
        }

        func factorial(n: int): int {
            var result: int = 1;
            for i := 1; i <= n; i := i + 1 {
                result := result * i;
            }
            return result;
        }

        func isPrime(x: int): bool {
            if x <= 1 {
                return false;
            }
            for i := 2; i < x; i := i + 1 {
                if x % i == 0 {
                    return false;
                }
            }
            return true;
        }

        func printPrimes(n: int): void {
            for i := 2; i <= n; i := i + 1 {
                if self.isPrime(i) {
                    io.@writeInt(i);
                }
            }
        }
    }
