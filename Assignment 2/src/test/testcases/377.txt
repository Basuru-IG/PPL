
    class Color {
        const RED: int = 1;
        const GREEN: int = 2;
        const BLUE: int = 3;
        
        func getColorName(color: int): string {
            if color == RED {
                return "Red";
            } 
            
            if color == GREEN {
                return "Green";
            }
            
            if color == BLUE {
                return "Blue";
            }
            return "Unknown";
        }
    }
