
    class Math {
        var PI: float = 3.14159;
        
        func square(x: float): float {
            return x * x;
        }
        
        func power(base: float, exp: float): float {
            var result: float = 1.0;
            for i := 0; i < exp; i := i + 1 {
                result := result * base;
            }
            return result;
        }
    }
