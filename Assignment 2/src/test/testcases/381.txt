
    class Shapes {
        func areaOfCircle(radius: float): float {
            var pi: float = 3.14159;
            return pi * radius * radius;
        }
        
        func areaOfRectangle(length: float, width: float): float {
            return length * width;
        }
        
        func volumeOfCylinder(radius: float, height: float): float {
            return self.areaOfCircle(radius) * height;
        }
    }
