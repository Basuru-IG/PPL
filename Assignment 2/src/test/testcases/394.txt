
    class ArrayManipulation {
        var arr: [5]int;
        
        func initializeArray(): void {
            for i := 0; i < 5; i := i + 1 {
                arr[i] := i * 2;
            }
        }
        
        func printArray(): void {
            for i := 0; i < 5; i := i + 1 {
                io.@writeInt(arr[i]);
            }
        }
    }
