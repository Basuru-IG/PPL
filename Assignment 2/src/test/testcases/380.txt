
    class Sorting {
        func bubbleSort(arr: ArrayType, n: int): void {
            var i: int = 0;
            for i := 0; i < n - 1; i := i + 1 {
                for j := 0; j < n - 1; j := j + 1 {
                    if arr[i] > arr[j] {
                        var temp: int = arr[i];
                        arr[i] := arr[j];
                        arr[j] := temp;
                    }
                }
            }
        }
    }
