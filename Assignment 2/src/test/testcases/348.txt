
class Salary {
    const @taxPercentage: float = 0.1;
    func scale(): float {}
}

class Salary <- JuniorSalary {
    func scale(): float {
        return 1.2;
    }
}

class Person {
    var base: float;
    
    func constructor(base: float, salary: Salary) {
        self.base := base;
        self.salary := salary;
    }
    
    func getSalary(): float {
        return (1 - Salary.@taxPercentage) * self.base * self.salary.scale;
    }
}
