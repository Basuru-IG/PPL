
    class ComplexMath {
        func fibonacci(n: int): int {
            if n <= 1 {
                return n;
            }
            return self.fibonacci(n - 1) + self.fibonacci(n - 2);
        }
        
        func gcd(a: int, b: int): int {
            if b == 0 {
                return a;
            }
            return self.gcd(b, a % b);
        }
        
        func lcm(a: int, b: int): int {
            return (a * b) / self.gcd(a, b);
        }
    }
